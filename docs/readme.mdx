import { Playground, PropsTable } from 'docz'
import { Value } from 'react-powerplug';
import { TMask } from '../src'
import { numberFormat } from './format';
import { css } from 'emotion';


## Basic usage

It can be used as number format

<Playground>
  <Value initial={''}>
    {text => (
      <TMask
        value={text.value}
        onChange={text.set}
        format={numberFormat}
      >
        {({ value, onChange }) => (
          <input value={value} onChange={onChange} className={css({textAlign:"right"})}/>
        )}
      </TMask>
    )}
  </Value>
</Playground>

Where `numberFormat` function is

```javascript
const numberFormat = str =>
  [...str.replace(/[^\d]+/gi, '')].reduce(
    (r, v, i) => ((i + 1) % 3 === 0 ? r + v + ' ' : r + v),
    ''
  );
```
