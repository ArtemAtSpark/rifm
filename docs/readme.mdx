import { Playground, PropsTable } from 'docz'
import { Value } from 'react-powerplug';
import { TMask } from '../dist/tmask.esm'


## Basic usage

It can be used as number format

<Playground>
  <Value initial={''}>
    {text => (
      <TMask
        value={text.value}
        onChange={text.set}
        format={str =>
          [...str.replace(/\s+/gi, '')].reduce(
            (r, v, i) => ((i + 1) % 3 === 0 ? r + v + ' ' : r + v),
            '',
          )
        }
      >
        {({ value, onChange }) => (
          <input value={value} onChange={onChange} />
        )}
      </TMask>
    )}
  </Value>
</Playground>
